
-- 2. Categories Table
CREATE TABLE Categories (
    CategoryID INT PRIMARY KEY IDENTITY(1,1),
    Name VARCHAR(100) NOT NULL,
    Description NVARCHAR(MAX)
);

-- 3. Products Table (Master)
CREATE TABLE Products (
    ProductID INT PRIMARY KEY IDENTITY(1,1),
    Name NVARCHAR(200) NOT NULL,
    Description NVARCHAR(MAX),
    Brand NVARCHAR(100),
    ReferenceLink NVARCHAR(MAX),
    PurchaseLocation NVARCHAR(200),
    -- Status 'Active', 'Out of Stock', 'On Demand'
    Status NVARCHAR(50) DEFAULT 'Active',
    CategoryID INT FOREIGN KEY REFERENCES Categories(CategoryID),
    CreatedAt DATETIME2 DEFAULT GETDATE()
);

-- 4. Inventory and Costs Table
-- Manages current stock and historical pricing
CREATE TABLE Inventory (
    InventoryID INT PRIMARY KEY IDENTITY(1,1),
    ProductID INT FOREIGN KEY REFERENCES Products(ProductID),
    PurchaseCost DECIMAL(18, 2) NOT NULL,
    SuggestedRetailPrice DECIMAL(18, 2) NOT NULL,
    CurrentStock INT DEFAULT 0,
    LastRestockDate DATETIME2 DEFAULT GETDATE(),
    LastSaleDate DATETIME2
);

-- 5. Tag Management (Many-to-Many)
CREATE TABLE Tags (
    TagID INT PRIMARY KEY IDENTITY(1,1),
    TagName VARCHAR(50) UNIQUE NOT NULL
);

CREATE TABLE Product_Tags (
    ProductID INT FOREIGN KEY REFERENCES Products(ProductID),
    TagID INT FOREIGN KEY REFERENCES Tags(TagID),
    PRIMARY KEY (ProductID, TagID)
);

-- 6. Promotions Table
CREATE TABLE Promotions (
    PromoID INT PRIMARY KEY IDENTITY(1,1),
    ProductID INT FOREIGN KEY REFERENCES Products(ProductID),
    DiscountPercentage DECIMAL(5, 2), -- e.g., 15.50 for 15.5%
    StartDate DATETIME2 NOT NULL,
    EndDate DATETIME2 NOT NULL,
    IsActive AS (CASE WHEN GETDATE() BETWEEN StartDate AND EndDate THEN 1 ELSE 0 END)
);
GO

-- 7. PERFORMANCE REPORT VIEW PROFITS AND AGING
CREATE VIEW ViewPerformanceReport AS
SELECT 
    p.Name,
    i.CurrentStock,
    i.PurchaseCost,
    i.SuggestedRetailPrice,
    (i.SuggestedRetailPrice - i.PurchaseCost) AS EstimatedProfit,
    CASE 
        WHEN i.PurchaseCost  0 THEN ((i.SuggestedRetailPrice - i.PurchaseCost)  i.PurchaseCost)  100 
        ELSE 0 
    END AS ProfitMarginPercentage,
    DATEDIFF(DAY, i.LastRestockDate, GETDATE()) AS DaysInInventory,
    p.Status
FROM Products p
JOIN Inventory i ON p.ProductID = i.ProductID;
GO

-- 8. Product Images Table
CREATE TABLE Product_Images (
    ImageID INT PRIMARY KEY IDENTITY(1,1),
    ProductID INT FOREIGN KEY REFERENCES Products(ProductID),
    ImagePath NVARCHAR(MAX) NOT NULL, -- e.g., 'imagesproductsitem-01.jpg'
    IsPrimary BIT DEFAULT 0,          -- 1 for cover, 0 for gallery
    DisplayOrder INT DEFAULT 0,       -- To determine gallery sequence
    RegisteredAt DATETIME2 DEFAULT GETDATE()
);
GO
